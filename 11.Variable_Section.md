myfirstplaybook.yml


```yaml
--- # My First YAML playbook for ansible
- hosts: apacheweb
  user: ansible
  sudo: yes
  connection: ssh
  gather_facts: no
  vars:
    playbook_version: 0.1b
  vars_files:
    - conf/copyright.yml
    - conf/webdefaults.yml
  vars_prompt:
    - name: web domain
      prompt: Web Domain
```
